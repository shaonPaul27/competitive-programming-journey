#define MinThree(a,b,c) min(a,min(b,c)) //finding min between 3 numbers

#define ll long long
#define LL long long
using namespace std;

void solve(){
    int n, c,l=0,r=0;
    cin>>n>>c;
    string s; cin>>s;
    for(int i=0; i<=c-1; i++)
        if(s[i] == 'R')l++;
    for(int i=c-1; i<s.size(); i++)
        if(s[i] == 'L') r++;

    cout<<min(l,r);
}
int main(){
    ll testCase = 1;
    cin>>testCase;
    for(ll test = 1; test<= testCase; test++){
        solve();
        cout<<"\n";
    }
    return 0;
}

problem 4: Substring Shift
sol:

#include <bits/stdc++.h>
#define MaxThree(a,b,c) max(a,max(b,c)) //finding max between 3 numbers
#define MinThree(a,b,c) min(a,min(b,c)) //finding min between 3 numbers
#define debug(x) cout<<#x<<": "<<x<<endl
#define DEBUG(x) cout<<#x<<": "<<x<<endl
#define ll long long
#define LL long long
using namespace std;

void solve(){
    int n; cin>>n;
    string s;  cin>>s;
    bool isFirst = true, allz = true;
    int cnt = 0;
    for(int i=0; i<n; i++){
        if(s[i]=='z' and isFirst == true){
            isFirst = false;
            cout<<'a';
            for(int j=i+1; j<n; j++){
                if(s[j]=='z'){
                    cout<<'a';
                    cnt++;
                }
                else{
                    i = i + cnt;
                    allz = false;
                    break;
                }
            }
            if(allz) return;
        }
        else cout<<s[i];
    }
}
int main(){
    ll testCase = 1;
    cin>>testCase;
    for(ll test = 1; test<= testCase; test++){
        solve();
        cout<<"\n";
    }
    return 0;
}
